// switch
#include <iostream>
using namespace std;
int main() 
{
    int day;
    cout << "enter the number (1-7) corresponding to the day of the week :";
    cin >>day;
switch (day)
{
    case 1:
        cout<<"monday"<<endl;
        break;
    case 2:
        cout<<"tusday"<<endl;
        break;
    case 3:
        cout<<"wenesday"<<endl;
        break;
    case 4:
        cout<<"thursday"<<endl;
        break;
    case 5:
        cout<<"friday"<<endl;
        break;
    case 6:
        cout<<"saturday"<<endl;
        break;
    case 7:
        cout<<"sunday"<<endl;
        break;
    default:
        cout<<"invalid input enter number between (1 to 7)"<<endl;
        break;
}
    return 0;
}

// switch
#include <iostream>
using namespace std;
int main() 
{
    char day;
    cout << "enter the day (s,m,t,w,h,f,y) corresponding to the day of the week :";
    cin >>day;
switch (day)
{
    case 'm':
        cout<<"1"<<endl;
        break;
    case 't':
        cout<<"2"<<endl;
        break;
    case 'w':
        cout<<"3"<<endl;
        break;
    case 'h':
        cout<<"4"<<endl;
        break;
    case 'f':
        cout<<"5"<<endl;
        break;
    case 'y':
        cout<<"6"<<endl;
        break;
    case 's':
        cout<<"7"<<endl;
        break;
    default:
        cout<<"invalid input enter number between (1 to 7)"<<endl;
        break;
}
    return 0;
}
